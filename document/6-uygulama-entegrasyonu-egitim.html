1 NuGet Paketlerini Yükleme

cd RestApi/RestApi

# OpenTelemetry Temel
dotnet add package OpenTelemetry --version 1.10.0
dotnet add package OpenTelemetry.Extensions.Hosting --version 1.10.0

# OpenTelemetry Traces
dotnet add package OpenTelemetry.Instrumentation.AspNetCore --version 1.10.1
dotnet add package OpenTelemetry.Instrumentation.Http --version 1.10.1
dotnet add package OpenTelemetry.Exporter.OpenTelemetryProtocol --version 1.10.0

# OpenTelemetry Metrics
dotnet add package OpenTelemetry.Instrumentation.Runtime --version 1.10.0
dotnet add package OpenTelemetry.Exporter.Prometheus.AspNetCore --version 1.10.0-rc.1

# Serilog
dotnet add package Serilog.AspNetCore
dotnet add package Serilog.Sinks.Elasticsearch
dotnet add package Serilog.Enrichers.Environment
dotnet add package Serilog.Enrichers.Process
dotnet add package Serilog.Enrichers.Thread
dotnet add package Serilog.Formatting.Elasticsearch

# Health Checks
dotnet add package AspNetCore.HealthChecks.Sqlite
⚠️ Not: Paket versiyonları zaman içinde değişebilir. dotnet add package komutu en son stabil versiyonu otomatik yükler. Prometheus exporter'ı şu an RC (Release Candidate) sürümündedir.

2 Program.cs — Tam Entegrasyon

using Microsoft.EntityFrameworkCore;
using RestApi.Data;
using Microsoft.AspNetCore.Authentication.JwtBearer;
using Microsoft.IdentityModel.Tokens;
using System.Text;
using System.Diagnostics;
using System.Diagnostics.Metrics;
using Serilog;
using Serilog.Sinks.Elasticsearch;
using Serilog.Formatting.Elasticsearch;
using OpenTelemetry.Trace;
using OpenTelemetry.Metrics;
using OpenTelemetry.Resources;
using Microsoft.AspNetCore.Diagnostics.HealthChecks;

// ========================================
// SERILOG YAPILANDIRMASI
// ========================================
Log.Logger = new LoggerConfiguration()
    .MinimumLevel.Information()
    .MinimumLevel.Override("Microsoft", Serilog.Events.LogEventLevel.Warning)
    .MinimumLevel.Override("System", Serilog.Events.LogEventLevel.Warning)
    .Enrich.FromLogContext()
    .Enrich.WithEnvironmentName()
    .Enrich.WithMachineName()
    .Enrich.WithProcessId()
    .Enrich.WithThreadId()
    .WriteTo.Console()
    .WriteTo.Elasticsearch(new ElasticsearchSinkOptions(
        new Uri("http://localhost:9200"))
    {
        IndexFormat = $"restapi-logs-{DateTime.UtcNow:yyyy.MM.dd}",
        AutoRegisterTemplate = true,
        AutoRegisterTemplateVersion = AutoRegisterTemplateVersion.ESv8,
        CustomFormatter = new ElasticsearchJsonFormatter(),
        BatchPostingLimit = 50,
        Period = TimeSpan.FromSeconds(2),
    })
    .CreateLogger();

try
{
    Log.Information("Starting RestApi application");

    var builder = WebApplication.CreateBuilder(args);

    // Serilog'u host'a bağla
    builder.Host.UseSerilog();

    // ========================================
    // OPENTELEMETRY YAPILANDIRMASI
    // ========================================
    var serviceName = "RestApi";
    var serviceVersion = "1.0.0";

    var resourceBuilder = ResourceBuilder.CreateDefault()
        .AddService(serviceName: serviceName, serviceVersion: serviceVersion)
        .AddAttributes(new Dictionary<string, object>
        {
            ["deployment.environment"] = builder.Environment.EnvironmentName
        });

    builder.Services.AddOpenTelemetry()
        // TRACES
        .WithTracing(tracing =>
        {
            tracing
                .SetResourceBuilder(resourceBuilder)
                .AddSource("RestApi.*")
                .AddAspNetCoreInstrumentation(options =>
                {
                    options.Filter = (httpContext) =>
                        !httpContext.Request.Path.StartsWithSegments("/health") &&
                        !httpContext.Request.Path.StartsWithSegments("/metrics");
                })
                .AddHttpClientInstrumentation()
                .AddOtlpExporter(options =>
                {
                    options.Endpoint = new Uri("http://localhost:14317");
                });
        })
        // METRICS
        .WithMetrics(metrics =>
        {
            metrics
                .SetResourceBuilder(resourceBuilder)
                .AddMeter("RestApi.*")
                .AddAspNetCoreInstrumentation()
                .AddHttpClientInstrumentation()
                .AddRuntimeInstrumentation()
                .AddPrometheusExporter();
        });

    // ========================================
    // UYGULAMA SERVİSLERİ
    // ========================================
    builder.Services.AddDbContext<ApplicationDbContext>(options =>
        options.UseSqlite(builder.Configuration.GetConnectionString("DefaultConnection")));
    builder.Services.AddControllers();
    builder.Services.AddSingleton<RestApi.Services.PasswordManager>();

    // Health Checks
    builder.Services.AddHealthChecks()
        .AddCheck("self", () =>
            Microsoft.Extensions.Diagnostics.HealthChecks.HealthCheckResult.Healthy("Application is running"));

    // JWT Authentication
    var jwtKey = builder.Configuration.GetValue<string>("Jwt:Key");
    if (string.IsNullOrEmpty(jwtKey))
        throw new InvalidOperationException("JWT Key is not configured");

    var key = Encoding.ASCII.GetBytes(jwtKey);
    builder.Services.AddAuthentication(JwtBearerDefaults.AuthenticationScheme)
        .AddJwtBearer(options =>
        {
            options.Events = new JwtBearerEvents
            {
                OnMessageReceived = context =>
                {
                    if (context.Request.Cookies.ContainsKey("X-Access-Token"))
                        context.Token = context.Request.Cookies["X-Access-Token"];
                    return Task.CompletedTask;
                }
            };
            options.RequireHttpsMetadata = false;
            options.SaveToken = true;
            options.TokenValidationParameters = new TokenValidationParameters
            {
                ValidateIssuerSigningKey = true,
                IssuerSigningKey = new SymmetricSecurityKey(key),
                ValidateIssuer = false,
                ValidateAudience = false
            };
        });

    // ========================================
    // UYGULAMA PIPELINE
    // ========================================
    var app = builder.Build();

    if (app.Environment.IsDevelopment())
        app.UseDeveloperExceptionPage();

    // Serilog request logging
    app.UseSerilogRequestLogging(options =>
    {
        options.MessageTemplate =
            "HTTP {RequestMethod} {RequestPath} responded {StatusCode} in {Elapsed:0.0000} ms";
    });

    app.UseMiddleware<GlobalMiddleware>();
    app.UseAuthentication();
    app.UseAuthorization();

    app.MapControllers();

    // Prometheus metrics endpoint
    app.MapPrometheusScrapingEndpoint();

    // Health check endpoint
    app.MapHealthChecks("/health");

    app.Run();
}
catch (Exception ex)
{
    Log.Fatal(ex, "Application terminated unexpectedly");
}
finally
{
    Log.CloseAndFlush();
}

4 prometheus.yml Güncelleme

# prometheus.yml
global:
  scrape_interval: 5s
  evaluation_interval: 5s

scrape_configs:
  - job_name: 'prometheus'
    static_configs:
      - targets: ['prometheus:9090']

  - job_name: 'restapi'
    static_configs:
      - targets: ['host.docker.internal:5000']
    metrics_path: '/metrics'
    scheme: 'http'
💡 Not: Eğer .NET uygulaması farklı bir portta çalışıyorsa, targets değerini güncelleyin. dotnet run varsayılan olarak 5000 (HTTP) ve 5001 (HTTPS) portlarını kullanır.

5 otel-config.yaml Güncelleme

# otel-config.yaml
receivers:
  otlp:
    protocols:
      grpc:
        endpoint: 0.0.0.0:4317
      http:
        endpoint: 0.0.0.0:4318

processors:
  batch:
    timeout: 5s
    send_batch_size: 1024

exporters:
  debug:
    verbosity: detailed

  otlp:
    endpoint: jaeger:4317
    tls:
      insecure: true

service:
  pipelines:
    traces:
      receivers: [otlp]
      processors: [batch]
      exporters: [debug, otlp]

    metrics:
      receivers: [otlp]
      processors: [batch]
      exporters: [debug]

    logs:
      receivers: [otlp]
      processors: [batch]
      exporters: [debug]

6 Entegrasyon Doğrulama Adımları

Docker Servislerini Başlatma

cd RestApi
docker compose down      # Önceki container'ları temizle
docker compose up -d     # Servisleri başlat
docker compose ps        # Durumu kontrol et

.NET Uygulamasını Başlatma

cd RestApi/RestApi
dotnet restore
dotnet run

Doğrulama Kontrolleri

# 1. Health Check
curl http://localhost:5000/health
# Beklenen: Healthy

# 2. Metrics Endpoint
curl http://localhost:5000/metrics
# Beklenen: Prometheus format metrikleri

# 3. API İstekleri (trace & log üretir)
curl http://localhost:5000/api/products

# 4. Prometheus Targets
# http://localhost:9090/targets
# restapi target'ı UP olmalı

# 5. Jaeger Traces
# http://localhost:16686
# Service: RestApi seçip trace'leri görüntüleyin

# 6. Elasticsearch Logs
curl "http://localhost:9200/restapi-logs-*/_search?q=*&pretty&size=5"

# 7. Kibana
# http://localhost:5601
# Data view oluşturun: restapi-logs-*

# 8. Grafana
# http://localhost:3000
# Prometheus data source ekleyin: http://prometheus:9090
✅ Tüm kontroller başarılı ise: Tüm monitoring alt yapısı hazır! Traces → Jaeger, Metrics → Prometheus/Grafana, Logs → Elasticsearch/Kibana.

🔧 Sık Karşılaşılan Sorunlar ve Çözümleri

Prometheus "restapi" target DOWN görünüyor

# .NET uygulamasının çalıştığını doğrulayın
curl http://localhost:5000/metrics

# Docker'dan host erişimini test edin
docker exec prometheus wget -qO- http://host.docker.internal:5000/metrics

# Linux kullanıyorsanız docker-compose'a network_mode ekleyin
# veya prometheus.yml'de IP adresini güncelleyin

Jaeger'da trace görünmüyor

# OTel Collector loglarını kontrol edin
docker logs otel-collector

# Collector'ın çalıştığını doğrulayın
docker compose ps otel-collector

# .NET'in OTLP endpoint'ine bağlanabildiğini kontrol edin
# Program.cs'deki endpoint: http://localhost:14317

Elasticsearch'te log yok

# Elasticsearch'ın çalıştığını kontrol edin
curl http://localhost:9200/_cluster/health?pretty

# Index'leri listeleyin
curl http://localhost:9200/_cat/indices?v

# Serilog konsol çıktısını kontrol edin
# Hata mesajları Elasticsearch bağlantı sorununu gösterir
